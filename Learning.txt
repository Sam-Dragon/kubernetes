## EXPORT & IMPORT RESOURCE [YAML]
kubectl create namespace ckad-namespace --dry-run=client -o yaml > ckad-namespace.yaml
kubectl create deployment hello-deployment --image=hello-world -n ckad-namespace --dry-run=client -o yaml > hello-deployment.yaml
... ADD OTHER RESOURCES AS WELL

kubectl create -f ckad-namespace.yaml,hello-deployment.yaml

## NAMESPACE
> GET / CREATE / DELETE / [EDIT - NOT ALLOWED]
kubectl get namespaces [Or namespace]
kubectl create namespace ckad-namespace
kubectl create namespace delete-keyspace
kubectl delete namespace delete-keyspace

> DEFAULT NAMESPACE FOR CONTEXT
kubectl config set-context rahul-cluster --namespace=ckad-namespace

## DEPLOYMENT
> GET / CREATE / DELETE / EDIT 
** [without namespace --> Default namespace]
> kubectl get deployments [Or deployment Or deploy]
** [with namespace]
kubectl get deployments -n ckad-namespace
kubectl create deployment hello-deployment --image=hello-world -n ckad-namespace
kubectl delete deploy hello-deployment -n ckad-namespace
kubectl edit deployment hello-deployment -n ckad-namespace

> DESCRIBE / SCALE [**SPECIAL**]
kubectl describe deployment hello-deployment -n ckad-namespace
kubectl scale deployment hello-deployment -n ckad-namespace --replicas=2

## PODS
> GET / CREATE / DELETE / EDIT 
** [without namespace --> Default namespace]
kubectl get pods [Or pod Or po]
** [with namespace]
kubectl get pods -n ckad-namespace
kubectl run nginx --image=nginx -n ckad-namespace
kubectl delete pods hello-deployment-6896f5bbfb-4sf84 -n ckad-namespace
kubectl edit pods hello-deployment-6896f5bbfb-m74zx -n ckad-namespace

> DESCRIBE / EXEC / LOGS / WIDE [**SPECIAL**]
kubectl describe pods hello-deployment-6896f5bbfb-m74zx -n ckad-namespace
kubectl exec -it hello-deployment-6896f5bbfb-m74zx -n ckad-namespace sh [Or  bash] -- Or cqlsh [Cassandra]
kubectl logs hello-deployment-6896f5bbfb-m74zx -n ckad-namespace
kubectl get pods -n ckad-namespace -o wide 

> WATCH
kubectl get pods -w
kubectl logs hello-deployment-6896f5bbfb-m74zx -n ckad-namespace -f

> VERBOSE
kubectl get po -n ckad-namespace -v=8

> EXPORT
kubectl run hello-world --image=hello-world -n ckad-namespace --dry-run=client -o yaml > hello-world.yaml

> CREATE WITH OPTIONS [PORT, ENV, LABELS]
kubectl run nginx --image=nginx -n ckad-namespace --port=5701 --env="APP_VERSION=1.0" --labels="app=ckad,env=prod"

> DRY RUN [TO Prevent component creation]
kubectl run hello-world --image=hello-world -n ckad-namespace --dry-run=client

## SERVICE
> GET / CREATE / DELETE / EDIT 
** [without namespace --> Default namespace]
kubectl get services [Or service Or svc]
** [with namespace]
kubectl get service -n ckad-namespace
kubectl create service clusterip hello-service -n ckad-namespace --clusterip="None" -o yaml --dry-run=client | kubectl set selector --local -f - 'environment=qa' -o yaml | kubectl create -f -
kubectl delete service hello-service -n ckad-namespace
kubectl edit service hello-service -n ckad-namespace

> DESCRIBE [**SPECIAL**]
kubectl describe service hello-service -n ckad-namespace
