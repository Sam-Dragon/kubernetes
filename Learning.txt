## Commands

## HELP
kubectl -h

## VERSION
kubectl version

## CLUSTER
kubectl cluster-info

## NAMESPACE
> GET / CREATE / DELETE / [EDIT - NOT ALLOWED]
kubectl get namespaces [Or namespace]
kubectl create namespace ckad-keyspace
kubectl create namespace delete-keyspace
kubectl delete namespace delete-keyspace

## DEPLOYMENT
> GET / CREATE / DELETE / EDIT 
** [without namespace --> Default namespace]
> kubectl get deployments [Or deployment Or deploy]
** [with namespace]
kubectl get deployments -n ckad-keyspace
kubectl create deployment hello-deployment --image=hello-world -n ckad-keyspace
kubectl delete deploy hello-deployment -n ckad-keyspace
kubectl edit deployment hello-deployment -n ckad-keyspace

> DESCRIBE / SCALE [**SPECIAL**]
kubectl describe deployment hello-deployment -n ckad-keyspace
kubectl scale deployment hello-deployment -n ckad-keyspace --replicas=2

## PODS
> GET / CREATE / DELETE / EDIT 
** [without namespace --> Default namespace]
kubectl get pods [Or pod Or po]
** [with namespace]
kubectl get pods -n ckad-keyspace
kubectl run nginx --image=nigix
kubectl delete pods hello-deployment-6896f5bbfb-4sf84 -n ckad-keyspace
kubectl edit pods hello-deployment-6896f5bbfb-m74zx -n ckad-keyspace

> DESCRIBE / EXEC / LOGS / WIDE [**SPECIAL**]
kubectl describe pods hello-deployment-6896f5bbfb-m74zx -n ckad-keyspace
kubectl exec -it hello-deployment-6896f5bbfb-m74zx -n ckad-keyspace sh [Or  bash] -- Or cqlsh [Cassandra]
kubectl logs hello-deployment-6896f5bbfb-m74zx -n ckad-keyspace
kubectl get pods -n ckad-keyspace -o wide 

> WATCH
kubectl get pods -w
kubectl logs hello-deployment-6896f5bbfb-m74zx -n ckad-keyspace -f

## REPLICA SET [SHOULD NOT BE USED, DEPLOYMENT IS TAKING CARE OF THIS]
> GET / CREATE - NOT ALLOWED / DELETE / EDIT 
** [without namespace --> Default namespace]
kubectl get replicasets [Or replicaset Or rs]
** [with namespace]
kubectl get replicasets -n ckad-keyspace
kubectl delete replicasets hello-deployment-6896f5bbfb -n ckad-keyspace
kubectl edit replicasets hello-deployment-6896f5bbfb -n ckad-keyspace

> DESCRIBE [**SPECIAL**]
kubectl describe replicasets hello-deployment-6896f5bbfb -n ckad-keyspace

## SERVICE
> GET / CREATE / DELETE / EDIT 
** [without namespace --> Default namespace]
kubectl get services [Or service Or svc]
** [with namespace]
kubectl get service -n ckad-keyspace
kubectl create service clusterip hello-service -n ckad-keyspace --clusterip="None" -o yaml --dry-run=client | kubectl set selector --local -f - 'environment=qa' -o yaml | kubectl create -f -
kubectl delete service hello-service -n ckad-keyspace
kubectl edit service hello-service -n ckad-keyspace

> DESCRIBE [**SPECIAL**]
kubectl describe service hello-service -n ckad-keyspace
