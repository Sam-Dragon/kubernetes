# Limit Range
- It allows us to configure
  1. Default memory and cpu
  2. Minimum memory and cpu
  3. Maximum memory and cpu

> mem-imit-range.yaml
vi mem-limit-range.yaml
---
apiVersion: v1
kind: LimitRange
metadata:
  name: mem-limit-range
spec:
  limits:
  - default:
      memory: 512Mi
    defaultRequest:
      memory: 256Mi
    max:
      memory: 1Gi
    min:
      memory: 50Mi
    type: Container
---
> copy and save
kubectl create -f mem-limit-range.yaml (OR) kubectl apply -f mem-limit-range.yaml

# Pod
kubectl run po nginx-mem --image=nginx -n ckad 


> cpu-limit-range.yaml
vi cpu-limit-range.yaml
---
apiVersion: v1
kind: LimitRange
metadata:
  name: cpu-limit-range
spec:
  limits:
  - default:
      cpu: 800m
    defaultRequest:
      cpu: 800m
    max:
      cpu: "800m"
    min:
      cpu: "200m" 
    type: Container
---
> copy and save
kubectl create -f cpu-limit-range.yaml (OR) kubectl apply -f cpu-limit-range.yaml

# Pod
kubectl run po nginx-cpu --image=nginx -n ckad
