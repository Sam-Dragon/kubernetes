# Description
limitranges shortform is 'limit'

# Create
kubectl create -f limit-range.yaml 

# Edit
kubectl edit limitranges default-max-minlimit-range -n ckad

# Get
kubectl get limitranges -n ckad

> Get all limitranges
kubectl get limitranges -n ckad

# Delete
kubectl delete limitranges default-max-minlimit-range -n ckad

# Describe
kubectl describe limitranges default-max-minlimit-range -n ckad

# Export 
> YAML
kubectl get limitranges create-lr -n ckad --dry-run=client -o yaml > limit-range.yaml
kubectl create -f config.yaml (OR) kubectl apply -f limit-range.yaml
